'use strict';
/* global define */
define([
    // Libraries.
    'jquery',
    'underscore',
    'backbone',

    'corejs/engine/logger',

    // Templates
    'templates/jst',

    // Plugins.
    'backbone.marionette',
    'corejs/vendor/backbone.analytics'
],

function(
    $,
    _,
    Backbone,
    logger,
    jst
) {

    /**
     * Provide a global location to place configuration settings and module creation
     * @exports app
     * @namespace
     */
    var app = {
        /**
         * The root path to run the application.
         * @type {String}
         * @default window.location.pathname
         */
        root: window.location.pathname,
        /**
         * Array with compiled templates
         * @type {Object}
         * @namespace
         */
        jst: jst,
        log: logger
    };

    return _.extend(new Backbone.Marionette.Application(), app);

});
