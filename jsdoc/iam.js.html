'use strict';
/* global define */
define([
    'corejs/app',
    'corejs/engine/silkroad/services',
    'corejs/engine/silkroad/common',
    'corejs/engine/jwt',
    'corejs/engine/validate',
    'underscore'
], function(app, services, common, jwt, validate, _) {

    var iam = {};

    iam.token = function() {
        return new TokenBuilder();
    };

    function TokenBuilder() {
        this.uri = 'oauth/token';
    }

    TokenBuilder.prototype.create = function(params) {
        // we need params to create access token
        validate.isValue(params, 'Create token request must contains params');
        // if there are oauth params this mean we should do use the GET verb
        if (params.oauth) {
            return doGetTokenRequest(this.uri, params);
        }
        // otherwise we use the traditional POST verb.
        return doPostTokenRequest(this.uri, params);
    };

    iam.user = function(id) {
        if (id) {
            return new UserBuilder(id);
        }
        return new UsersBuilder();
    };

    function UsersBuilder() {
        this.uri = 'user';
    }

    /**
     * Creates a new user.
     * @param  {Object} data the user data.
     * @return {Promise}     a promise which resolves into the ID of the created user or fails with a {SilkRoadError}.
     */
    UsersBuilder.prototype.create = function(data) {
        return services.requestXHR({
            url: buildUri(this.uri),
            method: services.method.POST,
            data: data,
            withAuth: true
        }).then(function(res) {
            return services.extractLocationId(res);
        });
    };

    UsersBuilder.prototype.getMe = function() {
        return getUser(services.method.GET, this.uri, 'me');
    };

    UsersBuilder.prototype.get = function(params) {
        return services.request({
            url: buildUri(this.uri),
            method: services.method.GET,
            query: params ? common.serializeParams(params) : null,
            withAuth: true
        });
    };

    function UserBuilder(id) {
        this.uri = 'user';
        this.id = id;
    }

    UserBuilder.prototype.get = function() {
        return getUser(services.method.GET, this.uri, this.id);
    };

    UserBuilder.prototype.update = function(data) {
        return services.request({
            url: buildUri(this.uri, this.id),
            method: services.method.PUT,
            data: data,
            withAuth: true
        });
    };

    UserBuilder.prototype.delete = function() {
        return services.request({
            url: buildUri(this.uri, this.id),
            method: services.method.DELETE,
            withAuth: true
        });
    };

    UserBuilder.prototype.addIdentity = function(identity) {
        validate.isValue(identity, 'Missing identity');
        return services.request({
            url: buildUri(this.uri, this.id) + '/identity',
            method: services.method.POST,
            data: identity,
            withAuth: true
        });
    };

    var buildUri = function(uri, id) {
        if (id) {
            uri += '/' + id;
        }
        return app.common.get('iamEndpoint') + uri;
    };

    var doGetTokenRequest = function(uri, params) {
        return services.request({
            url: buildUri(uri),
            method: services.method.GET,
            query: $.param(_.extend({
                    assertion: getJwt(params),
                    'grant_type': app.common.get('grantType')
                },
                params.oauth))
            });
    };

    var doPostTokenRequest = function(uri, params) {
        return services.request({
            url: buildUri(uri),
            method: services.method.POST,
            data: {
                assertion: getJwt(params),
                'grant_type': app.common.get('grantType')
            },
            contentType: 'application/x-www-form-urlencoded; charset=UTF-8'
        });
    };

    var getJwt = function(params) {
        if (params.jwt) {
            return params.jwt;
        }
        if (params.claims) {
            return jwt.generate(params.claims);
        } else {
            throw 'Create token request must contains either jwt or claims parameter';
        }
    };

    var getUser = function(method, uri, id) {
        return services.request({
            url: buildUri(uri, id),
            method: services.method.GET,
            withAuth: true
        });
    };

    return iam;
});
