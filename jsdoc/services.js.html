'use strict';
/* global define */
define([
    'corejs/engine/silkroad/silkRoadError',
    'corejs/engine/services',
    'q',
    'underscore'
], function(silkRoadError, services, q, _) {

    var silkRoadServices = {};
    // is an extension of services
    _.extend(silkRoadServices, services);

    //request is overwritten
    /**
     * SilkRoad server request. This function handles backend errors and maps them to {SilkRoadError}.
     *
     * @param  {Object}  args  The request argms as required by corejs/engine/services
     * @param  {Boolean} isXHR If set to true the returning promise results in a low-level object containing: data, textStatus and xhr. Otherwise, it resolves with just the data.
     * @return {Promise}    a {Q} promise
     */
    silkRoadServices.request = function(args, isXHR) {
        var defer = q.defer();
        services.request(args).then(
            function(data, textStatus, jqXHR) {
                defer.resolve(isXHR ? {
                    data: data,
                    textStatus: textStatus,
                    xhr: jqXHR
                } : data);
            },
            function(jqXHR, textStatus, errorThrown) {
                if (jqXHR.status) {
                    defer.reject(silkRoadError.create(jqXHR));
                } else {
                    defer.reject(textStatus + (errorThrown ? (':' + errorThrown) : ''));
                }
            });
        return defer.promise;
    };

    silkRoadServices.requestXHR = function(args) {
        return this.request(args, true);
    };

    /**
     * Extract a id from the location header of a requestXHR
     * @param  {Promise} res response from a requestXHR
     * @return {String}  id from the Location
     */
    silkRoadServices.extractLocationId = function(res) {
        var uri = res.xhr.getResponseHeader('Location');
        return uri ? uri.substr(uri.lastIndexOf('/') + 1) : undefined;
    };

    return silkRoadServices;
});
